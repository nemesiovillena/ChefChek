# Organización de Procedimientos

## Resumen

Sistema estructurado para organizar, categorizar y mantener procedimientos operativos de cocina. Incluye jerarquía de documentos, plantillas estandarizadas, etiquetado inteligente y asignación de responsables.

### Generación Automática de Etiquetas

```typescript
function generateAutoTags(document: WikiDocumentDto): string[] {
  const tags: string[] = [];

  // Tag by category
  tags.push(document.category.toLowerCase());

  // Tag by status
  tags.push(document.status.toLowerCase());

  // Tag by creation date
  const ageDays = Math.floor((Date.now() - new Date(document.createdAt).getTime()) / (1000 * 60 * 60 * 24));
  if (ageDays < 30) tags.push('reciente');
  if (ageDays > 365) tags.push('histórico');

  // Tag by view count
  if (document.viewCount > 100) tags.push('popular');
  if (document.viewCount > 500) tags.push('muy_popular');

  // Tag by version
  if (document.version > 10) tags.push('versión_alta');

  return tags;
}
```

### Asignación de Responsable por Documento

```typescript
interface DocumentResponsible {
  documentId: string;
  responsibleUserId: string;
  responsibleRole: UserRole;
  backupResponsibleId?: string;
  backupResponsibleRole?: UserRole;
}

async function assignResponsible(
  documentId: string,
  dto: DocumentResponsible
): Promise<void> {
  const assignment = await this.documentResponsibleRepository.create({
    id: uuidv4(),
    ...dto,
    assignedAt: new Date(),
    assignedBy: userId,
  });

  await this.documentResponsibleRepository.save(assignment);

  // Notify responsible user
  await notifyUser(dto.responsibleUserId, {
    type: 'wiki_responsibility',
    documentId,
    message: `Has sido asignado como responsable del documento: ${documentTitle}`,
  });
}
```

## Workflow de Aprobación

### Estados de Aprobación

```typescript
enum ApprovalStatus {
  DRAFT = 'DRAFT',
  PENDING_REVIEW = 'PENDING_REVIEW',
  APPROVED = 'APPROVED',
  REJECTED = 'REJECTED',
  PUBLISHED = 'PUBLISHED',
}

interface ApprovalWorkflow {
  status: ApprovalStatus;
  reviewers: string[];
  approvals: number;
  requiredApprovals: number;
  rejections: number;
}
```

### Proceso de Aprobación

```typescript
async function submitForApproval(documentId: string, userId: string): Promise<void> {
  const document = await this.getDocument(documentId, tenantId, userId);

  document.status = WikiStatus.PENDING_REVIEW;
  document.updatedBy = userId;
  await this.wikiDocumentRepository.save(document);

  // Create approval workflow
  const workflow = this.approvalWorkflowRepository.create({
    id: uuidv4(),
    documentId,
    status: ApprovalStatus.PENDING_REVIEW,
    reviewers: await getApprovers(document.category, tenantId),
    approvals: 0,
    requiredApprovals: 2, // Requires 2 approvals
    rejections: 0,
  });

  await this.approvalWorkflowRepository.save(workflow);

  // Notify reviewers
  for (const reviewerId of workflow.reviewers) {
    await notifyUser(reviewerId, {
      type: 'wiki_approval_request',
      documentId,
      documentTitle: document.title,
      message: `El documento "${document.title}" requiere tu aprobación`,
    });
  }
}

async function approveDocument(documentId: string, userId: string): Promise<void> {
  const workflow = await this.approvalWorkflowRepository.findOne({
    where: { documentId, status: ApprovalStatus.PENDING_REVIEW },
  });

  if (!workflow) {
    throw new NotFoundException('Approval workflow not found');
  }

  if (!workflow.reviewers.includes(userId)) {
    throw new ForbiddenException('User is not an approver');
  }

  workflow.approvals += 1;
  await this.approvalWorkflowRepository.save(workflow);

  if (workflow.approvals >= workflow.requiredApprovals) {
    // All approvals received
    const document = await this.getDocument(documentId, tenantId, userId);
    document.status = WikiStatus.APPROVED;
    document.updatedBy = userId;
    await this.wikiDocumentRepository.save(document);

    workflow.status = ApprovalStatus.APPROVED;
    await this.approvalWorkflowRepository.save(workflow);

    // Notify creator
    await notifyUser(document.createdBy, {
      type: 'wiki_approved',
      documentId,
      message: `Tu documento "${document.title}" ha sido aprobado`,
    });
  }
}
```
